/* create a website that generates code from text */ var text = document.createElement('textarea'); text.setAttribute('id', 'text'); text.setAttribute('rows', '10'); text.setAttribute('cols', '50'); document.body.appendChild(text); var button = document.createElement('button'); button.innerHTML = 'Generate'; button.setAttribute('onclick', 'generate()'); document.body.appendChild(button); var code = document.createElement('div'); code.setAttribute('id', 'code'); document.body.appendChild(code); function generate() { var text = document.getElementById('text').value; var code = document.getElementById('code'); code.innerHTML = ''; var lines = text.split('\n'); for (var i = 0; i < lines.length; i++) { var line = lines[i]; var lineCode = document.createElement('div'); lineCode.innerHTML = line; code.appendChild(lineCode); } }
